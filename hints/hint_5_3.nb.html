Remove Words Example

#first see it without removals:
mydata_word_counts%>%
  anti_join(stop_words)%>%
  filter(n > 400) %>%
  mutate(word = reorder(word, n)) %>%
  ggplot(aes(n, word)) +
  geom_col() +
  labs(y = NULL)
Joining with `by = join_by(word)`

then try with the removed words:

#Example answer
mydata_word_counts%>%
  anti_join(stop_words)%>%
  filter(!(word %in% c("https", "t.co")))%>%
  filter(n > 400) %>%
  mutate(word = reorder(word, n)) %>%
  ggplot(aes(n, word)) +
  geom_col() +
  labs(y = NULL)
Joining with `by = join_by(word)`
